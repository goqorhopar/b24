# Пример файла конфигурации для Meeting Bot Server
# Скопируйте этот файл в .env и заполните ваши данные

# ===========================================
# ОБЯЗАТЕЛЬНЫЕ ПЕРЕМЕННЫЕ
# ===========================================

# Telegram Bot Token (получить у @BotFather)
TELEGRAM_BOT_TOKEN=your_telegram_bot_token_here

# Gemini AI API Key (получить в Google AI Studio)
GEMINI_API_KEY=your_gemini_api_key_here

# ===========================================
# BITRIX24 ИНТЕГРАЦИЯ (ОПЦИОНАЛЬНО)
# ===========================================

# ===========================================

# Уровень логирования (DEBUG/INFO/WARNING/ERROR)

# Порт для веб-сервера

# НАСТРОЙКИ ВСТРЕЧ
# ===========================================

# Режим без GUI (true для сервера)

# Максимальная длительность встречи в минутах

# НАСТРОЙКИ АУДИО
# ===========================================

# Частота дискретизации

# Количество каналов

# Размер чанка

# ===========================================
# НАСТРОЙКИ РАСПОЗНАВАНИЯ РЕЧИ
# ===========================================

# Модель Whisper (tiny/base/small/medium/large)
WHISPER_MODEL=base

# Язык распознавания
WHISPER_LANGUAGE=ru

# Устройство для вычислений (auto/cpu/cuda)
WHISPER_DEVICE=auto

# ===========================================
# НАСТРОЙКИ GEMINI AI
# ===========================================

# Основная модель

# Резервная модель

# Температура (креативность)

# ===========================================
# НАСТРОЙКИ CHROME/WEBDRIVER
# ===========================================

# Режим без GUI

# Размер окна

# User Agent

# ===========================================
# ПРОИЗВОДИТЕЛЬНОСТЬ
# ===========================================

# Максимальное количество одновременных встреч

# Таймаут встречи в секундах

# ===========================================

# Максимальное количество временных файлов

# ===========================================

# Chat ID администратора для уведомлений

# ===========================================

# Токен GitHub для автоматических коммитов

# Пользователь на сервере

ENABLE_DEPLOY_NOTIFICATIONS=true

# Включить уведомления об ошибках

# Включить уведомления о встречах
